SIMPLE INTEGRATION (Just Batching):

    from src.context_batcher import get_context_batcher

    batcher = get_context_batcher()

    batches, _ = batcher.add_queries_to_batches(batches, [query])

    for batch in batches:
        response = llm.query(batch.documents, batch.queries)


FULL INTEGRATION (Compression + Batching + Smart Retrieval):

    from src.context_window_manager import get_context_window_manager
    from src.document_compressor import get_document_compressor
    from src.context_batcher import get_context_batcher
    from src.smart_retrieval_long import get_smart_retriever

    cwm = get_context_window_manager()
    compressor = get_document_compressor()
    batcher = get_context_batcher()
    retriever = get_smart_retriever()

    # 1. Allocate
    allocation = cwm.allocate_context(documents, query)

    # 2. Compress
    compressed = compressor.compress_batch(documents, allocation)

    # 3. Select with Smart Retrieval
    selection = retriever.retrieve(documents, query, strategy="hybrid")

    # 4. Batch
    batches = batcher.pack_documents(selection.selected_documents)
    batches, _ = batcher.add_queries_to_batches(batches, [query])

    # 5. Process
    for batch in batches:
        response = llm.query(batch.documents, batch.queries)
